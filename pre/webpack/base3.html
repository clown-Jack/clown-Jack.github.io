 		return module.exports;
 	}

        //判断缓存区是否存在
        if (installedModules[moduleId]) {

(function(modules) {
    //设置缓存区
    var installedModules = [];
    //0是已经加载成功的代码块
    var installedChunks = {
        main: 0
    }

    function webpackJsonpCallback(data) { //data = []
        var chunkIds = data[0]; //["title"]
        var moreModules = data[1];
        let resolves = [];
        for (let i = 0; i < chunkIds.length; i++) {
            let chunkId = chunkIds[i];
            resolves.push(installedChunks[chunkId][0]); //installedChunks[chunkId]=[resolve,reject,promise]
            installedChunks[chunkId] = 0; //0是已经加载成功的代码块
        }
        for (let moduleId in moreModules) {
            modules[moduleId] = moreModules[moduleId]
        }
        //保留老的push方法  如果不bind的话data不会放到数组里，this指向window
        if (parentJsonpFunction) parentJsonpFunction(data);

        while (resolves.length) { //数组的shift方法表示取出第一个元素
            resolves.shift()();
        }
    }

    function __webpack_require__(moduleId) {
        //判断缓存区是否存在
        if (installedModules[moduleId]) {
            return installedModules[moduleId].exports
        }
        //如果缓存区不存在 添加到缓存区
        var module = installedModules[moduleId] = {
            i: moduleId,
            l: false, //表示是否加载，如果没有加载false 加载过true

        module.l = true; //表示加载过

        return module.exports

    }
    //mode为什么要用二进制来判断 linux 权限7 111 可读可写可执行

const babel = require("@babel/core");


//loader参数 第一个是要转化的内容  第二个是它的sourceMap
//this.request = loader1!loader2!loader3!xxx.css  我们要的是xxx.js
function loader(source, sourceMap) {
    const options = {
        presets: ['@babel/preset-env'], //babel的预设
        inputSourceMap: sourceMap, //输入的sourceMap
        sourceMaps: true, //告诉webpack我要输出souceMap
        filename: this.request.split('!').pop() //指定的文件
    }
    let { code, map, ast } = babel.transform(source, options);
    //code 是转译后的代码  map是sourceMap  ast抽象语法树
    return this.callback(null, code, map, ast);
}

/**
 * inline pre post normal
 * eslint  代码风格检查  一般用于前置
 */

let path = require("path");
//正常情况下module的查找路径
let nodeModules = path.resolve(__dirname, "node_modules");
//行内的loader
let request = '-!inline-loader1!inline-loader2!./style.css';
let rules = [
    { test: /\.css$/, enforce: "pre", use: ['pre-loader1', 'pre-loader2'] },
    { test: /\.css$/, enforce: "normal", use: ['normal-loader1', 'normal-loader2'] }, //默认是normal类的loader
    { test: /\.css$/, enforce: "post", use: ['post-loader1', 'post-loader2'] }
]

let resolveLoader = loader => path.resolve(nodeModules, loader + "js"); //相对路径变成绝对路径

//不要前置和普通的loader 只剩下后置和行内
const noPreAutoLoaders = request.startsWith("-!");
//不要普通loader
const noAutoLoaders = request.startsWith("!");
//只要行内
const noPrePostAutoLoaders = request.startsWith("!!")
let inlineLoaders = request.replace(/^-?!+/, "").replace(/!!+/g, "!").split("!");
// let inlineLoaders = request.split("!"); //["inline-loader1","inline-loader2","./style.css"]
let resource = inlineLoaders.pop(); //截取最后一个


let preLoaders = [];
let postLoaders = [];
let normalLoaders = [];
for (let i = 0; i < rules.length; i++) {
    let rule = rules[i];
    if (rule.test.test(resource)) { //绝对路径是否和每个规则里的test匹配
        if (rule.enforce == "pre") {
            preLoaders.push(...rule.use)
        } else if (rule.enforce == 'post') {
            postLoaders.push(...rule.use)
        } else {
            normalLoaders.push(...rule.use)
        }
    }
}
//合并loader的执行顺序  从右到左执行
let loaders;
if (noPreAutoLoaders) {
    loaders = [...postLoaders, ...inlineLoaders];
} else if (noPrePostAutoLoaders) {
    loaders = [...inlineLoaders];
} else if (noAutoLoaders) {
    loaders = [...postLoaders, ...inlineLoaders, ...preLoaders];
} else {
    loaders = [...postLoaders, ...inlineLoaders, ...normalLoaders, ...preLoaders];
}
//经过这个映射，把一个loader模块名变成一个绝对路径
loaders = loaders.map(resolveLoader)
console.log(loaders);

//loader-running.js
const path = require("path");
const fs = require("fs");

function createLoaderObject(loader) {
    let loaderObj = { data: {} };
    loaderObj.request = loader; //loader文件的绝对路径
    loaderObj.normal = require(loader);
    loaderObj.pitch = loaderObj.normal.pitch;
    return loaderObj
}

function runLoaders(options, callback) {
    let loaderContext = {}; //这个对象将会成为loader函数中的this
    let resource = options.resource; //要加载的资源
    let loaders = options.loaders; //["loader1","loader2","loader3"]
    loaders = loaders.map(createLoaderObject); //[{narmal,pitch},{narmal,pitch},{narmal,pitch}]
    loaderContext.loaderIndex = 0; //当前的索引
    loaderContext.readResource = fs; //读取文件
    loaderContext.resource = resource; //index.js
    loaderContext.loaders = loaders; //[{narmal,pitch},{narmal,pitch},{narmal,pitch}]
    iteratePitchingLoader(loaderContext, callback);

    function processResource(loaderContext, callback) {
        let buffer = loaderContext.readResource.readFileSync(loaderContext.resource, "utf-8");
        loaderContext.loaderIndex--;
        iterateNormalLoader(loaderContext, buffer, callback);
    }

    function iterateNormalLoader(loaderContext, args, callback) {
        if (loaderContext.loaderIndex < 0) {
            return callback(null, args);
        }
        let currentLoaderObject = loaderContext.loaders[loaderContext.loaderIndex];
        let normalFn = currentLoaderObject.normal;
        let result = normalFn.call(loaderContext, args);
        loaderContext.loaderIndex--;
        iterateNormalLoader(loaderContext, result, callback)
    }

    function iteratePitchingLoader(loaderContext, callback) {
        if (loaderContext.loaderIndex >= loaderContext.loaders.length) {
            return processResource(loaderContext, callback);
        }
        let currentLoaderObject = loaderContext.loaders[loaderContext.loaderIndex];
        let pitchFn = currentLoaderObject.pitch;
        if (!pitchFn) {
            loaderContext.loaderIndex++;
            iteratePitchingLoader(loaderContext, callback);
        }
        let result = pitchFn.apply(loaderContext);
        if (result) { //如果这个result有值就会发生反转
            loaderContext.loaderIndex--;
            iterateNormalLoader(loaderContext, result, callback); //执行normalLoader函数pitch返回的结果要向下传
        } else {
            loaderContext.loaderIndex++;
            iteratePitchingLoader(loaderContext, callback);
        }
    }
}

let entry = './src/index.js';

let options = {
    resource: path.resolve(__dirname, entry),
    loaders: [
        path.resolve(__dirname, "loaders/loader1.js"),
        path.resolve(__dirname, "loaders/loader2.js"),
        path.resolve(__dirname, "loaders/loader3.js"),
    ]
}

runLoaders(options, (err, result) => {
    console.log("执行完毕");
    console.log(result);
})
